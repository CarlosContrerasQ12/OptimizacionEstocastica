Si el avión intenta evitar una región que llamaremos $R$ y queremos que llegue en el menor tiempo posible a una region objetivo $O$, entonces podríamos modelar los costos del problema como

\begin{equation} r((z,w)|((x,y),a))= \begin{cases} -1 & \text{ si } (z,w)\notin R \cup O\\ -20 & \text{ si } (w,w) \in R\\ 0 & \text{ si } (z,w) \in O \end{cases} \end{equation}

Así, el avión estará incentivado a ir a la region objetivo evitando fuertemente entrar en la región restringida. Y si consideramos un factor de descuento suficientemente grande, entonces la recompensa total promedio va a ser mayor si se pasa más tiempo en la región objetivo, y esto pasa si se llega más rápido a esta, por lo que la politica óptima minimizará tiempos esperados de llegada.

    end

Iteración por política¶

Vamos a programar primero el algoritmo de iteración por política. Para ello, es necesario poder construir las matrices de transición dada una política. Por la formulación de este problema, esta matriz tendrá ceros en la mayoria de sus componentes, por lo que se puede guardar eficientemente en memoria en un formato sparse (CSC es el default en Julia). También será necesarió construir un vector de recompensas esperadas para cada estado.

function contruirP(S,acciones,estados_alcanzables,probabilidades_alcanzables,conjuntoProbs,pol)
    N=size(S)[1]
    P = SparseMatrixDict(N,N)
    for i in 1:N
        ind=indexin( pol[i], acciones[i] )[1]
        for (indj,j) in enumerate(estados_alcanzables[i][ind])
            P[i,j]=conjuntoProbs[probabilidades_alcanzables[i][ind]][indj]
        end
    end
    return sparse(P)
end

function linear_programming_tulip(λ,κ,S,acciones,estados_alcanzables,probabilidades_alcanzables,conjuntoProbs,R,O)
    model = Model(Tulip.Optimizer)
    JuMP.set_optimizer_attribute(model, "IPM_IterationsLimit", 200)
    set_silent(model)
    N=length(S)
    total=0
    indices=[]
    @variable(model, v[1:N])
    @constraint(model,c[i=1:length(S),inda=1:length(acciones[i])],v[i]-sum(conjuntoProbs[probabilidades_alcanzables[i][inda]][indj]*(λ*v[j]+recom(j,R,O)) for (indj,j) in enumerate(estados_alcanzables[i][inda]))>=0)
    @objective(model, Min, κ'*v)
    #println(model)
    optimize!(model)
    medidas=[]
    for i in 1:length(S)
        mes=[]
        for j in 1:length(acciones[i])
            append!(mes,dual(c[i,j]))
        end
        append!(medidas,[mes])
    end
    return model,medidas
end

function linear_programming_Mosek(λ,κ,S,acciones,estados_alcanzables,probabilidades_alcanzables,conjuntoProbs,R,O)
    model = Model(Mosek.Optimizer)
                        
    set_optimizer_attribute(model, "MSK_DPAR_OPTIMIZER_MAX_TIME", 100.0)
    set_optimizer_attribute(model, "MSK_DPAR_BASIS_TOL_S", 1e-6)
    set_optimizer_attribute(model, "MSK_IPAR_OPTIMIZER", MSK_SOLVE_DUAL)
    #set_optimizer_attribute(model, "MSK_IPAR_OPTIMIZER", 0)
    set_silent(model)
    N=length(S)
    total=0
    indices=[]
    @variable(model, v[1:N])
    @constraint(model,c[i=1:length(S),inda=1:length(acciones[i])],v[i]-sum(conjuntoProbs[probabilidades_alcanzables[i][inda]][indj]*(λ*v[j]+recom(j,R,O)) for (indj,j) in enumerate(estados_alcanzables[i][inda]))>=0)
    @objective(model, Min, κ'*v)
    #println(model)
    optimize!(model)
    medidas=[]
    for i in 1:length(S)
        mes=[]
        for j in 1:length(acciones[i])
            append!(mes,dual(c[i,j]))
        end
        append!(medidas,[mes])
    end
    return model,medidas
end

Conclusiones¶

  • Ambos metodos funcionan eficientemente para hallar políticas óptimas. Si se modela un sistema con espacio de estados moderado y se dispone de un solver de programación lineal muy eficiente, este método puede ser más rápido que el método de iteración por política programado sin optimizar profesionalmente código. En este caso es mejor confiar en las grandes compañias productoras de software.
  • El método de iteración por política parece no sufrir tanto con $\lambda$ más cerca a 1, a comparación del método de iteración por valor que se implementó en la tarea pasada, en donde el número de iteraciones escalaba considerablemente cuando se hacia $\lambda$ más cercano a 1.
  • El método de programación lineal aparenta no ser suceptible a la escogencia de la distribución $\kappa$, como se espera en la teoría planteada. Sin embargo, habría que realizar más pruebas para ver si afecta en algo su escogencia en la práctica (tal vez algunas escogencias hagan el sistema lineal más fácil de resolver, por ejemplo, no sé).
  • Para obtener caminos eficientes que minimizen el tiempo de llegada se requieren $\lambda$ suficientemente cercanos a 1.
