Si el avin intenta evitar una regin que llamaremos $R$ y queremos que llegue en el menor tiempo posible a una region objetivo $O$, entonces podramos modelar los costos del problema como

\begin{equation} r((z,w)|((x,y),a))= \begin{cases} -1 & \text{ si } (z,w)\notin R \cup O\\ -20 & \text{ si } (w,w) \in R\\ 0 & \text{ si } (z,w) \in O \end{cases} \end{equation}

As, el avin estar incentivado a ir a la region objetivo evitando fuertemente entrar en la regin restringida. Y si consideramos un factor de descuento suficientemente grande, entonces la recompensa total promedio va a ser mayor si se pasa ms tiempo en la regin objetivo, y esto pasa si se llega ms rpido a esta, por lo que la politica ptima minimizar tiempos esperados de llegada.

                estados_alcanzables[n]=[ad,ar]
                probabilidades_alcanzables[n]=[1 ,1]

    end

Iteración por política¶

Vamos a programar primero el algoritmo de iteracin por poltica. Para ello, es necesario poder construir las matrices de transicin dada una poltica. Por la formulacin de este problema, esta matriz tendr ceros en la mayoria de sus componentes, por lo que se puede guardar eficientemente en memoria en un formato sparse (CSC es el default en Julia). Tambin ser necesari construir un vector de recompensas esperadas para cada estado.

In [6]:
function contruirP(S,acciones,estados_alcanzables,probabilidades_alcanzables,conjuntoProbs,pol)
    N=size(S)[1]
    P = SparseMatrixDict(N,N)
    for i in 1:N
        ind=indexin( pol[i], acciones[i] )[1]
        for (indj,j) in enumerate(estados_alcanzables[i][ind])
            P[i,j]=conjuntoProbs[probabilidades_alcanzables[i][ind]][indj]
        end
    end
    return sparse(P)
end

function contruir_r(S,R,O,pol)
    N=size(S)[1]
    r=Vector{Float64}(undef,N)
    for i in 1:N
        #ind=indexin( pol[i], acciones[i] )[1]
        r[i]=recom(i,R,O)
    end
    return r
end

function iteracion_por_politica(pol0,,S,acciones,estados_alcanzables,probabilidades_alcanzables,conjuntoProbs,R,O)
    N=size(S)[1]
    p0=Vector{Char}(undef,N)
    p0=copy(pol0)
    p1=Vector{Char}(undef,N)
    v0=zeros(N)
    N_iter=0
    println("Comienza a calcular")
    while true
        Pn=contruirP(S,acciones,estados_alcanzables,probabilidades_alcanzables,conjuntoProbs,p0)
        M=sparse(I,N,N)-(*Pn)
        r=contruir_r(S,R,O,p0)
        rE=Pn*r
        vn=M\rE
        for i in 1:N
            pso=[]
            for (inda,a) in enumerate(acciones[i])
                suma=0
                cog=conjuntoProbs[probabilidades_alcanzables[i][inda]]
                for (inde,est) in enumerate(estados_alcanzables[i][inda])
                    suma+=cog[inde]*(*vn[est]+recom(est,R,O))
                end
                append!(pso,suma)
            end
            
            if (abs.(maximum(pso) .- pso).<1e-3)[indexin(p0[i],acciones[i])[1]]==true
                p1[i]=p0[i]
            else
                p1[i]=acciones[i][argmax(pso)]
            end
            if N_iter>70 && p1[i]!=p0[i]
                println(pso)
            end
        end
        N_iter+=1
        #println("Iteraciones: ",N_iter, " diferencias: ",sum(1 .- (p1.==p0)) ," Error en v: ",maximum(abs.(vn-v0)))
        if (p1==p0)
            println("Convergi en ",N_iter," iteraciones")
            return p1,p0
        end
        if (N_iter>100)
            return p1,p0
        end
        p0=copy(p1)
        v0=copy(vn)
    end
    
end

Programación lineal¶

Ahora formulemos el problema como un problema de optimizacin lineal. Para esto vamos a usar JuMP, el lenguaje de modelacin para problemas de optimacin de Julia. Vamos tambin a usar el solver Tulip, que soporta usa mtodos de punto interior con modificaciones que resuelven el problema dual simultaneamente.

In [17]:

Es ms fcil introducir las restricciones del problema primal partiendo de la informacin del modelo que tenemos. Por lo tanto, formulamos el primal y el solver de Tulip halla simultaneamente la solucin del primal y el dual, que es la que nos interesa para obtener las medidas de ocupacin y con ellas construir las polticas ptimas.

In [18]:
function linear_programming_tulip(,,S,acciones,estados_alcanzables,probabilidades_alcanzables,conjuntoProbs,R,O)
    model = Model(Tulip.Optimizer)
    JuMP.set_optimizer_attribute(model, "IPM_IterationsLimit", 200)
    set_silent(model)
    N=length(S)
    total=0
    indices=[]
    @variable(model, v[1:N])
    @constraint(model,c[i=1:length(S),inda=1:length(acciones[i])],v[i]-sum(conjuntoProbs[probabilidades_alcanzables[i][inda]][indj]*(*v[j]+recom(j,R,O)) for (indj,j) in enumerate(estados_alcanzables[i][inda]))>=0)
    @objective(model, Min, '*v)
    #println(model)
    optimize!(model)
    medidas=[]
    for i in 1:length(S)
        mes=[]
        for j in 1:length(acciones[i])
            append!(mes,dual(c[i,j]))
        end
        append!(medidas,[mes])
    end
    return model,medidas
end

Conclusiones¶

  • Ambos metodos funcionan eficientemente para hallar polticas ptimas. Si se modela un sistema con espacio de estados moderado y se dispone de un solver de programacin lineal muy eficiente, este mtodo puede ser ms rpido que el mtodo de iteracin por poltica programado sin optimizar profesionalmente cdigo. En este caso es mejor confiar en las grandes compaias productoras de software.
  • El mtodo de iteracin por poltica parece no sufrir tanto con $\lambda$ ms cerca a 1, a comparacin del mtodo de iteracin por valor que se implement en la tarea pasada, en donde el nmero de iteraciones escalaba considerablemente cuando se hacia $\lambda$ ms cercano a 1.
  • El mtodo de programacin lineal aparenta no ser suceptible a la escogencia de la distribucin $\kappa$, como se espera en la teora planteada. Sin embargo, habra que realizar ms pruebas para ver si afecta en algo su escogencia en la prctica (tal vez algunas escogencias hagan el sistema lineal ms fcil de resolver, por ejemplo, no s).
  • Para obtener caminos eficientes que minimizen el tiempo de llegada se requieren $\lambda$ suficientemente cercanos a 1.
